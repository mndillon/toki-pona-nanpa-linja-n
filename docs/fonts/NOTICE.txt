Using the nasin-nanpa-5.0.0-beta.3-UCSUR-nanpa-linja-n-v5.otf font in Python:

#import functionality from PIL
from PIL import Image, ImageDraw, ImageFont

#font file name
SITELEN_NANPA_LINJA_N_FONT = "nasin-nanpa-5.0.0-beta.3-UCSUR-nanpa-linja-n-v5.otf"

#define font (with RAQM renderer)
sitelen_nanpa_linja_n_font = ImageFont.truetype(
        str(SITELEN_NANPA_LINJA_N_FONT),
        size=72,
        layout_engine=ImageFont.Layout.RAQM
    )

#Unicode code points for sitelen cartouche start and end
CARTOUCHE_START_UCSUR = chr(0xF1990)
CARTOUCHE_END_UCSUR   = chr(0xF1991)

# OpenType features (safe fallback below if unsupported)
_OT_FEATURES = ["calt", "liga", "ccmp"]

#draw text (Unicode sitelen code points) on to image using font, fill=(0,0,0,255)
def _safe_draw_text(draw: ImageDraw.ImageDraw, xy: Tuple[int, int], s: str, font: ImageFont.FreeTypeFont, fill) -> None:
    if not s:
        return
    try:
        draw.text(xy, s, font=font, fill=fill, features=_OT_FEATURES)
    except TypeError:
        draw.text(xy, s, font=font, fill=fill)
